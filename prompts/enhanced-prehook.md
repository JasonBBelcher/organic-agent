{
    public Post $post;

    public string $title = '';

    public string $content = '';
    
    public function save()
    {
        $this->validate();
        
        $this->post->update([
            'title' => $this->title,
            'content' => $this->content,
        ]);
        
        $this->dispatch('post-saved');
        
        return redirect()->route('posts.index');
    }
    
    public function render()
    {
        return view('livewire.post-form');
    }
}
